📦 dict_cache.db — 主数据库 (72KB, WAL模式)

TABLE: dict_info (词典元信息)

idINTEGER PK AUTOINCREMENT
pathTEXT UNIQUE
nameTEXT
mdd_pathTEXT
priorityINTEGER DEFAULT 0

TABLE: standard_entries (词条存储)

wordTEXT
contentBLOB (zlib压缩HTML)
dict_idINTEGER → dict_info.idFK
lengthINTEGER
INDEX: idx_word(COLLATE NOCASE), idx_dict

VIRTUAL TABLE: fts_entries (FTS5全文搜索)

wordTEXT
content_textTEXT (纯文本)
dict_idUNINDEXED

TABLE: vocabulary (词汇本 - 艾宾浩斯)

wordTEXT PRIMARY KEY
added_timeREAL
review_stageINTEGER DEFAULT 0 (0-7)
next_review_timeREAL
contextTEXT
sourceTEXT
xpINTEGER DEFAULT 0

TABLE: search_history (搜索历史)

wordTEXT PRIMARY KEY
last_access_timeREAL
search_countINTEGER DEFAULT 1

TABLE: rss_sources (新闻源配置)

idINTEGER PK AUTOINCREMENT
nameTEXT
urlTEXT UNIQUE
enabledINTEGER DEFAULT 1
🎒 mdd_cache.db — 资源数据库 (16KB, WAL模式)

TABLE: resources (词典资源)

dict_idINTEGERPK part
norm_keyTEXT (规范化路径)PK part
dataBLOB (原始字节)
PRIMARY KEY: (dict_id, norm_key)
支持: 图片(.png/.jpg)、音频(.mp3/.wav)、CSS、字体等

TABLE: cached_dicts (缓存元信息)

dict_idINTEGER PRIMARY KEY
mdd_pathTEXT (来源路径)
cached_timeREAL (导入时间戳)
